// THE SPEC

One file to rule them all

ILP uses plain Markdown with a YAML frontmatter header. Familiar, readable, and tool-agnostic.

.ilp.md — example project instructions
---
ilp_version: "1.0"
name: "backend-api"
scope: "project"     # global | user | project | task
priority: 100        # higher = evaluated first
tags: ["typescript", "node", "api"]
uses:
  - skill: "python-best-practices"
    version: "^1.2"
---

# Project Instructions

## Code Style
- Use TypeScript strict mode at all times
- Prefer async/await over raw Promises
- All functions must have explicit return types

## Architecture
- Follow hexagonal architecture
- All side effects live in /src/adapters
- Use dependency injection via constructor params
// INTEGRATION

Plug into any agent

ILP resolves and merges instructions at runtime, then injects them as the agent's system prompt — before MCP tools are invoked.

agent.ts — ILP + MCP bootstrap
import Anthropic from "@anthropic-ai/sdk";
import { resolveILPContext } from "./ilp-parser";

const client = new Anthropic();

async function runAgent(userMessage: string) {
  // 1. Resolve ILP — merges global → user → project
  const ilp = await resolveILPContext(process.cwd());

  // 2. Inject ILP as system prompt
  const system = `You are an AI agent. Follow these instructions:

${ilp.instructions}`;

  // 3. Call model with MCP servers for tool access
  return client.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 8096,
    system,
    messages: [{ role: "user", content: userMessage }],
    mcp_servers: [
      { type: "url", url: "https://mcp.filesystem.local/sse", name: "fs" },
      { type: "url", url: "https://mcp.github.com/sse",        name: "github" },
    ],
  });
}
